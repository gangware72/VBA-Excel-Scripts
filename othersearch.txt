Sub SearchDataCompanyName()    Dim nameSearchFor As String    Dim ws As Worksheet    Dim lastRow As Long    Dim erow As Long    Dim count As Integer    Dim printMsg As String    Dim searchTerm, index As String    Dim hit As Long    Dim no As String        no = "NO"        lastRow = Sheets("Master List").Cells(Rows.count, "D").End(xlUp).Row 'finds the last row    count = 24                       For x = 3 To lastRow        'Make all comparators UPPER has same effect as IGNORECASE        searchTerm = UCase(Sheets("Search").Range("F5").Value) 'store searched term                If searchTerm = "" Then 'If the search term is empty set counter to end            x = lastRow        End If                index = UCase(Sheets("Master List").Cells(x, 4)) 'store company name at each line        hit = InStr(index, searchTerm) 'InStr returns the amount of occurences of index within searchTerm                If hit > 0 Then                    Sheets("Search").Cells(count, 2) = Sheets("Master List").Cells(x, 4) 'COMPANY NAME            Sheets("Search").Cells(count, 3) = Sheets("Master List").Cells(x, 7) 'EMAIL                        Sheets("Search").Cells(count, 5) = Sheets("Master List").Cells(x, 8).Value 'PASSWORD                        'Check if worksheet is returned and return the date it was or "NO"            If Sheets("Master List").Cells(x, 14).Value = "---" Then                Sheets("Search").Cells(count, 6) = no                        Else                Sheets("Search").Cells(count, 6) = Sheets("Master List").Cells(x, 14).Value 'worksheet returned                count = count + 2                            End If                        'Check if UPC's sent to client and return the date it was or "NO"            If Sheets("Master List").Cells(x, 15).Value = "---" Then                Sheets("Search").Cells(count, 7) = no                                        Else                Sheets("Search").Cells(count, 7) = Sheets("Master List").Cells(x, 15).Value 'UPC Sent to client                            End If                                                            End If                Next x                      End Sub
