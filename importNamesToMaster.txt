Sub importNamesToMaster()    'This macro is for importing manager scores into a master list containing the names of all employees    'Loop Variables    Dim i As Integer 'outer loop count for Import Employees tab    Dim j As Integer 'inner loop count for Employee Information        'Cell Variables    Dim importEmployeeCell As String 'current cell in Import Employees    Dim employeeInformation As String 'current cell in Employee Information        'Sheet Names    Dim importemp As String    Dim employeeinfo As String    importemp = "Import Employees"    employeeinfo = "Master Employee Information"        'last rows    Dim lastRowImportE, lastRowEmployeeinfo As Integer        lastRowImportE = 100 'last row for import employees    lastRowEmployeeinfo = 900 'last row for employee info        'value variables    Dim boxscore As Integer            'begin execution------------------------------------------------------------        For i = 3 To lastRowImportE            importEmployeeCell = "B" & i 'accesses the Import Employees worksheet                    For j = 5 To lastRowEmployeeinfo                    employeeInformation = "C" & j 'accesses the Employees worksheet                                    'if there is an name match then import the score to the master sheet            If Sheets(importemp).Range(importEmployeeCell).Value = Sheets(employeeinfo).Range(employeeInformation).Value Then                                 boxscore = Sheets(importemp).Range("C" & i).Value                                  Sheets(employeeinfo).Range("I" & j).Value = boxscore                                End If                                                            Next j            Next i                    End Sub
